New structure:
pythonclass RecommendationAgent(BaseAgent):  # Not BaseLangGraphAgent
    def __init__(self, current_user, db, ...):
        self.retrieval_agent = RetrievalAgent(...)
        self.curation_agent = CurationAgent()
    
    def execute(self, request: AgentRequest) -> AgentResponse:
        # Stage 1
        retrieval_response = self.retrieval_agent.execute(request)
        
        if not retrieval_response.book_recommendations:
            return self._no_candidates_fallback(request)
        
        # Stage 2
        final_response = self.curation_agent.execute(
            request=request,
            candidates=retrieval_response.book_recommendations,
            retrieval_context=retrieval_response.execution_state
        )
        
        return final_response

State additions:
pythondef _finalize_node(self, state):
    # Capture tool execution order and results
    tool_summary = []
    for exec in state.tool_executions:
        if exec.succeeded and self.tool_executor.is_book_recommendation_tool(exec.tool_name):
            book_count = len(self.tool_executor.extract_book_ids_from_result(exec.result))
            tool_summary.append({
                "tool": exec.tool_name,
                "arguments": exec.arguments,
                "book_count": book_count,
                "order": len(tool_summary) + 1
            })
    
    state.intermediate_outputs["retrieval_summary"] = tool_summary
    state.mark_completed()
    return state

            "author": book.author or "",

            "subjects": book.subjects or [],
            "tones": book.tones or [],
            "description": description,
            "genre": book.genre or "",
        })
    
    return prepared

Structure:
pythonclass RecommendationAgent(BaseAgent):
    def __init__(self, current_user, db, user_num_ratings, warm_threshold, allow_profile):
        self.retrieval_agent = RetrievalAgent(
            current_user=current_user,
            db=db,
            user_num_ratings=user_num_ratings,
            allow_profile=allow_profile,
        )
        self.curation_agent = CurationAgent()
    
    def execute(self, request: AgentRequest) -> AgentResponse:
        # Stage 1: Retrieval
        try:
            retrieval_response = self.retrieval_agent.execute(request)
        except Exception as e:
            append_chatbot_log(f"Retrieval failed: {e}")
            return self._fallback_response(request, "retrieval_error")
        
        candidates = retrieval_response.book_recommendations
        
        # Check if we got any candidates
        if not candidates:
            return self._no_candidates_response(request)
        
        append_chatbot_log(f"Retrieval: {len(candidates)} candidates from "
                          f"{len(retrieval_response.execution_state.tool_executions)} tools")
        
        # Stage 2: Curation
        try:
            retrieval_summary = retrieval_response.execution_state.intermediate_outputs.get(
                "retrieval_summary", []
            )
            
            final_response = self.curation_agent.execute(
                request=request,
                candidates=candidates,
                retrieval_summary=retrieval_summary
            )
            
            # Basic validation only
            if not final_response.book_recommendations:
                append_chatbot_log("Curation returned no books, using fallback")
                return self._fallback_with_candidates(request, candidates)
            
            return final_response
            
        except Exception as e:
            append_chatbot_log(f"Curation failed: {e}")
            return self._fallback_with_candidates(request, candidates)
Fallback handlers:
pythondef _no_candidates_response(self, request):
    return AgentResponse(
        text="I couldn't find books matching your request in our catalog (pre-2004). "
             "Try broader search terms or different subjects?",
        target_category="recsys",
        success=False,
    )

def _fallback_with_candidates(self, request, candidates):
    """Use when curation fails - return first 10 with generic prose."""
    return AgentResponse(
        text="Here are some books from our catalog that might interest you.",
        book_recommendations=candidates[:10],
        target_category="recsys",
        success=True,
    )

def _fallback_response(self, request, reason):
    """Use when retrieval fails."""
    append_chatbot_log(f"Complete fallback: {reason}")
    return AgentResponse(
        text="I had trouble processing your request. Please try rephrasing?",
        target_category="recsys",
        success=False,
    )
