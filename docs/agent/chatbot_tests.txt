3. AgentFactory - Add agent provider injection
python# app/agents/infrastructure/agent_factory.py

class AgentFactory:
    def __init__(
        self,
        # Add provider callables for each agent type
        recsys_provider: Optional[Callable[..., Agent]] = None,
        web_provider: Optional[Callable[..., Agent]] = None,
        docs_provider: Optional[Callable[..., Agent]] = None,
        respond_provider: Optional[Callable[..., Agent]] = None,
    ):
        # Use injected providers or default to real agent creation
        self._recsys_provider = recsys_provider or self._create_recsys
        self._web_provider = web_provider or self._create_web
        self._docs_provider = docs_provider or self._create_docs
        self._respond_provider = respond_provider or self._create_respond

    @staticmethod
    def _create_recsys(**kwargs) -> Agent:
        return RecommendationAgent(**kwargs)

    # ... similar for other agent types

    def create_agent(self, target: Target, **kwargs) -> Agent:
        if target == "recsys":
            return self._recsys_provider(**kwargs)
        elif target == "web":
            return self._web_provider(**kwargs)
        # ... etc
That's it. Production code unchanged, tests can inject mocks. No backward compatibility issues since defaults create real agents.
